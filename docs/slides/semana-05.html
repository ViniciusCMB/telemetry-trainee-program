Firmware Arduino — 20 Hz

const unsigned long PERIODO = 50;  // 20 Hz
unsigned long ultima_amostra = 0;
unsigned long contador = 0;

void setup() {
  Serial.begin(115200);
  Wire.begin();
  bmp.begin(0x76);
}

void loop() {
  unsigned long agora = millis();
  if (agora - ultima_amostra < PERIODO) return;
  ultima_amostra = agora;

  float temp = bmp.readTemperature();
  float pressao = bmp.readPressure();
  float alt = bmp.readAltitude(1013.25);
  contador++;

  enviarPacote(contador, alt, temp, pressao);
}

Função de Envio com Checksum

void enviarPacote(unsigned long id, float alt,
                  float temp, float pressao) {
  char buffer[128];
  int n = snprintf(buffer, sizeof(buffer),
    "$T,%lu,%.2f,%.2f,%.2f*",
    id, alt, temp, pressao);

  uint8_t cksum = 0;
  for (int i = 1; buffer[i] != '*'; i++) {
    cksum ^= buffer[i];
  }

  snprintf(buffer + n, sizeof(buffer) - n, "%02X\n", cksum);
  Serial.print(buffer);
}

Saída: $T,1,120.50,25.30,101325.00*3F

Python — Leitura da Serial

import serial
import time

def conectar(porta="/dev/ttyACM0", baud=115200):
    ser = serial.Serial(
        port=porta,
        baudrate=baud,
        timeout=0.1,  # 100 ms de timeout
    )
    time.sleep(2)  # aguarda reset do Arduino
    ser.reset_input_buffer()
    return ser

Timeout baixo permite loop responsivo.

Loop de Leitura em Tempo Real

def ler_serial(ser, callback):
    buffer = ""
    while True:
        try:
            char = ser.read(1).decode("utf-8", errors="replace")
            if char == "":
                continue

            buffer += char

            if char == "\n":
                linha = buffer.strip()
                buffer = ""
                if linha:
                    callback(linha)

        except serial.SerialException:
            print("Erro: conexao perdida")
            break
        except KeyboardInterrupt:
            print("\nEncerrando...")
            break

Pipeline de Processamento

def processar_linha(linha):
    if not linha.startswith("$T"):
        return None

    partes = linha[1:].split("*")
    if len(partes) != 2:
        return None

    dados, cksum_str = partes
    campos = dados.split(",")

    if len(campos) != 5:  # T,id,alt,temp,pressao
        return None

    cksum_calc = 0
    for c in dados:
        cksum_calc ^= ord(c)
    cksum_esperado = int(cksum_str, 16)

    if cksum_calc != cksum_esperado:
        print(f"Checksum invalido: {linha.strip()}")
        return None

    return {
        "id": int(campos[1]),
        "altitude": float(campos[2]),
        "temperatura": float(campos[3]),
        "pressao": float(campos[4]),
        "timestamp": time.time(),
    }

Logger CSV com Timestamp

import csv
from datetime import datetime

class CSVLogger:
    def __init__(self, nome_base="log_voo"):
        timestamp = datetime.now().strftime("%Y%m%d_%H%M%S")
        self.caminho = f"{nome_base}_{timestamp}.csv"
        self.arquivo = open(self.caminho, "w", newline="")
        self.escritor = csv.writer(self.arquivo)
        self.escritor.writerow([
            "timestamp_unix", "hora", "pacote_id",
            "altitude", "temperatura", "pressao"
        ])

    def escrever(self, dado):
        hora = datetime.fromtimestamp(
            dado["timestamp"]
        ).strftime("%H:%M:%S.%f")[:-3]
        self.escritor.writerow([
            f"{dado['timestamp']:.3f}",
            hora,
            dado["id"],
            f"{dado['altitude']:.2f}",
            f"{dado['temperatura']:.2f}",
            f"{dado['pressao']:.2f}",
        ])
        self.arquivo.flush()  # garante escrita em disco

    def fechar(self):
        self.arquivo.close()

Script Principal Unificado

def main():
    ser = conectar()
    logger = CSVLogger()
    pacotes_validos = 0
    pacotes_invalidos = 0

    def callback(linha):
        nonlocal pacotes_validos, pacotes_invalidos
        dado = processar_linha(linha)
        if dado:
            logger.escrever(dado)
            pacotes_validos += 1
            if pacotes_validos % 100 == 0:
                print(f"{pacotes_validos} pacotes coletados")
        else:
            pacotes_invalidos += 1

    try:
        ler_serial(ser, callback)
    finally:
        logger.fechar()
        ser.close()
        total = pacotes_validos + pacotes_invalidos
        print(f"\nTotal: {total} pacotes")
        print(f"Validos: {pacotes_validos}")
        print(f"Invalidos: {pacotes_invalidos}")
        print(f"Taxa: {pacotes_validos/(total or 1):.1%}")

Debug e Troubleshooting

  • Sem dados: Verificar porta e permissões (grupo dialout)
  • Dados corrompidos: Taxa de baud igual nos dois lados
  • Pacotes perdidos: buffer serial pequeno, aumentar timeout
  • Checksum errado: Verificar \r\n vs \n
  • Reset ao conectar: DTR enable, usar time.sleep(2)
# Descobrir porta
ls /dev/ttyACM* /dev/ttyUSB*
# Ver permissoes
ls -l /dev/ttyACM0
# Adicionar ao grupo
