Por que CSV na Telemetria?

CSV (Comma-Separated Values) é o formato mais comum para logs:

  • Leve, legível, universalmente suportado
  • Ferramentas de solo exportam telemetria em CSV
  • Pandas, Excel, Google Sheets abrem diretamente
  • Perfeito para análise pós-voo

Lendo CSV com DictReader

csv.DictReader mapeia cada linha para um dicionário:

import csv

def ler_dados(caminho):
    dados = []
    with open(caminho, mode='r') as arquivo:
        leitor = csv.DictReader(arquivo)
        for linha in leitor:
            dados.append(linha)
    return dados

# Uso
dados = ler_dados("log_voo.csv")
print(dados[0])  # primeira linha como dict

Cada linha vira um dict com as colunas como chaves.

Convertendo Tipos

CSV lê tudo como string — precisamos converter:

def converter_linha(linha):
    return {
        "timestamp": int(linha["timestamp"]),
        "altitude": float(linha["altitude"]),
        "temp": float(linha["temp"]),
        "pressao": float(linha["pressao"]),
        "voltage": float(linha["voltage"]),
    }

dados = [converter_linha(l) for l in ler_dados("log_voo.csv")]

Calculando Estatísticas

def calcular_estatisticas(dados, campo):
    valores = [linha[campo] for linha in dados]
    return {
        "media": sum(valores) / len(valores),
        "maximo": max(valores),
        "minimo": min(valores),
        "amostras": len(valores),
    }

stats_temp = calcular_estatisticas(dados, "temp")
print(f"Temperatura media: {stats_temp['media']:.2f} C")
print(f"Max: {stats_temp['maximo']:.2f} C")
print(f"Min: {stats_temp['minimo']:.2f} C")

Sempre valide se a lista não está vazia antes de dividir!

Salvando Resumo em Arquivo

def salvar_resumo(stats, nome_arquivo="resumo.txt"):
    with open(nome_arquivo, "w") as f:
        f.write("=== RESUMO DE TELEMETRIA ===\n\n")
        for campo, valores in stats.items():
            f.write(f"{campo}:\n")
            f.write(f"  Media:  {valores['media']:.2f}\n")
            f.write(f"  Max:    {valores['maximo']:.2f}\n")
            f.write(f"  Min:    {valores['minimo']:.2f}\n")
            f.write(f"  Amostras: {valores['amostras']}\n\n")
    print(f"Resumo salvo em {nome_arquivo}")

matplotlib — Primeiro Gráfico

import matplotlib.pyplot as plt

def plotar_altitude(dados):
    tempos = [linha["timestamp"] for linha in dados]
    altitudes = [linha["altitude"] for linha in dados]

    plt.figure(figsize=(10, 5))
    plt.plot(tempos, altitudes, color="#ff6b35", linewidth=2)
    plt.title("Altitude vs Tempo")
    plt.xlabel("Tempo (ms)")
    plt.ylabel("Altitude (m)")
    plt.grid(True, alpha=0.3)
    plt.tight_layout()
    plt.savefig("altitude.png", dpi=150)
    plt.show()

Subplots — Múltiplos Sensores

def plotar_multiplos(dados):
    fig, axs = plt.subplots(2, 2, figsize=(12, 8))
    tempo = [l["timestamp"] for l in dados]

    axs[0, 0].plot(tempo, [l["altitude"] for l in dados])
    axs[0, 0].set_title("Altitude")
    axs[0, 1].plot(tempo, [l["temp"] for l in dados], color="red")
    axs[0, 1].set_title("Temperatura")
    axs[1, 0].plot(tempo, [l["pressao"] for l in dados], color="green")
    axs[1, 0].set_title("Pressao")
    axs[1, 1].plot(tempo, [l["voltage"] for l in dados], color="orange")
    axs[1, 1].set_title("Voltagem")

    plt.tight_layout()
    plt.savefig("telemetria_completa.png")

Tratamento de Erros

Sempre considere que o arquivo pode estar ausente ou mal formatado:

def ler_dados_seguro(caminho):
    try:
        with open(caminho, mode='r') as f:
            return list(csv.DictReader(f))
    except FileNotFoundError:
        print(f"Erro: arquivo {caminho} nao encontrado")
        return []
    except csv.Error as e:
        print(f"Erro ao ler CSV: {e}")
        return []

Script Completo

import csv
import matplotlib.pyplot as plt

def main():
    dados = ler_dados_seguro("log_voo.csv")
    if not dados:
        return

    dados_convertidos = [converter_linha(l) for l in dados]

    stats = {}
    for campo in ["altitude", "temp", "pressao", "voltage"]:
        stats[campo] = calcular_estatisticas(dados_convertidos, campo)

    salvar_resumo(stats)
    plotar_altitude(dados_convertidos)
    plotar_multiplos(dados_convertidos)

if __name__ == "__main__":
    main()
