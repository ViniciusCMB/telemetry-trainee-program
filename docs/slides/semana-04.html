platformio.ini — Multi-Environments

[platformio]
default_envs = native

[env]
monitor_speed = 115200

[env:native]
platform = native
test_framework = unity

[env:uno]
platform = atmelavr
board = uno
framework = arduino
lib_deps =
  adafruit/Adafruit BMP280 Library@^2.6

[env:teensy]
platform = teensy

framework = arduino
lib_deps =
  adafruit/Adafruit BMP280 Library@^2.6

[env:esp32]
platform = espressif32
board = esp32-devkitc-v4
framework = arduino

lib_deps — Gerenciamento de Dependências

PlatformIO baixa e gerencia bibliotecas automaticamente:

[env:teensy]
lib_deps =
  adafruit/Adafruit BMP280 Library@^2.6.0
  adafruit/Adafruit BusIO@^1.14
  bblanchon/ArduinoJson@^7.0.0
  # Tambem aceita URLs de GitHub:
  # https://github.com/usuario/biblioteca.git

lib_ignore =
  Adafruit BusIO  # nao usar esta lib neste env

Nunca mais baixe ZIP manualmente!

build_flags e build_unflags

Configure defines e flags do compilador:

[env:teensy]
build_flags =
  -O2                        # Otimizacao
  -DDEBUG_LEVEL=2            # Define em C
  -DSERIAL_BAUD=115200
  -std=c++17

build_unflags =
  -O0                        # Remove flag padrao

[env:esp32]
build_flags =
  -DCORE_DEBUG_LEVEL=5       # Debug maximo
  -DBOARD_HAS_PSRAM

Use no código:

#ifdef DEBUG_LEVEL
  Serial.print("Debug: ");
  Serial.println(valor);
#endif

lib/ — Bibliotecas Internas

Cada biblioteca dentro de lib/ tem seu próprio namespace:

// lib/BMP280Wrapper/BMP280Wrapper.h
#pragma once
#include <Adafruit_BMP280.h>

class BMP280Wrapper {
public:
  bool begin(uint8_t addr = 0x76);
  float lerTemperatura();
  float lerPressao();
  float lerAltitude(float pressao_nivel_mar = 1013.25);
private:
  Adafruit_BMP280 bmp;
};
// lib/BMP280Wrapper/BMP280Wrapper.cpp
#include "BMP280Wrapper.h"

bool BMP280Wrapper::begin(uint8_t addr) {
  return bmp.begin(addr);
}

float BMP280Wrapper::lerTemperatura() {
  return bmp.readTemperature();
}

src/main.cpp — Código Principal

#include <Arduino.h>
#include "BMP280Wrapper.h"
#include "PacketProtocol.h"

BMP280Wrapper sensor;
PacketProtocol protocolo;
unsigned long ultima_amostra = 0;

void setup() {
  Serial.begin(115200);
  Wire.begin();

  if (!sensor.begin()) {
    Serial.println("Falha no sensor!");
    while (1);
  }
}

void loop() {
  unsigned long agora = millis();
  if (agora - ultima_amostra < 50) return;
  ultima_amostra = agora;

  float temp = sensor.lerTemperatura();
  float pressao = sensor.lerPressao();
  float alt = sensor.lerAltitude();

  protocolo.enviarPacoteTelemetria(agora, alt, temp, pressao);
}

Testes com Unity

Unity é um framework de testes para C/C++:

// test/test_checksum.cpp
#include <unity.h>
#include "PacketProtocol.h"

PacketProtocol protocolo;

void setUp(void) {}
void tearDown(void) {}

void test_checksum_correto() {
  uint8_t calc = protocolo.calcularChecksum("T,100,25.30,101325");
  uint8_t esperado = 'T' ^ ',' ^ '1' ^ '0' ^ '0' ^ ',' ... ;
  TEST_ASSERT_EQUAL_UINT8(esperado, calc);
}

void test_checksum_vazio() {
  uint8_t calc = protocolo.calcularChecksum("");
  TEST_ASSERT_EQUAL_UINT8(0, calc);
}

void setup() {
  UNITY_BEGIN();
  RUN_TEST(test_checksum_correto);
  RUN_TEST(test_checksum_vazio);
  UNITY_END();
}

void loop() {}

Rodando Testes

# Testes nativos (no PC, rapido)
pio test -e native

# Testes no hardware Teensy
pio test -e teensy

# Com verbose
pio test -e native -v

# Filtrar por nome
pio test -e native --filter checksum

Testes nativos compilam para seu PC e rodam em segundos.

Teste com Dados de Exemplo

// test/test_packet.cpp
void test_parse_pacote_valido() {
  const char* linha = "$T,100,120.50,25.30,101325,5.01*3F\n";

  Pacote pacote;
  bool ok = protocolo.parseLinha(linha, &pacote);

  TEST_ASSERT_TRUE(ok);
  TEST_ASSERT_EQUAL_INT(100, pacote.timestamp);
  TEST_ASSERT_FLOAT_WITHIN(0.01, 120.50, pacote.altitude);
  TEST_ASSERT_FLOAT_WITHIN(0.01, 25.30, pacote.temperatura);
  TEST_ASSERT_EQUAL_UINT8(0x3F, pacote.checksum);
}

void test_rejeita_checksum_invalido() {
  const char* linha = "$T,100,120.50*00\n";  // checksum errado
  Pacote pacote;
  bool ok = protocolo.parseLinha(linha, &pacote);
  TEST_ASSERT_FALSE(ok);
}

Integração Contínua (CI)

PlatformIO se integra com GitHub Actions:

# .github/workflows/test.yml
name: Test
on: [push, pull_request]
jobs:
  test:
    runs-on: ubuntu-latest
    steps:
      - uses: actions/checkout@v4
      - uses: actions/setup-python@v5
        with:
          python-version: '3.11'
      - name: Install PlatformIO
        run: pip install platformio
      - name: Run native tests
        run: pio test -e native
