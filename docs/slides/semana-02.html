Por que Parsing é Importante?

O Flight Computer envia dados como strings delimitadas:

$T,100,120.5,25.3,101325,5.01*3F
$T,200,245.1,24.8,100500,4.98*2D
$T,300,368.7,24.2,99500,4.95*1B
  • $T — tipo do pacote (telemetry)
  • 100,120.5,... — dados separados por vírgula
  • *3F — checksum hexadecimal

Estrutura do Pacote

Formato definido pelo protocolo de telemetria:

$<tipo>,<timestamp>,<altitude>,<temp>,<pressao>,<voltage>*<checksum>

Campos:

  • tipo: T (telemetria), E (erro), S (status)
  • timestamp: ms desde o boot
  • altitude: metros
  • temp: graus Celsius
  • pressao: Pascal
  • voltage: Volts
  • checksum: XOR de todos os bytes entre $ e *

Parser de Pacotes

def parse_packet(linha):
    if not linha.startswith("$"):
        return None

    # Remove $ e quebra no *
    conteudo, checksum_str = linha[1:].split("*")
    campos = conteudo.split(",")
    tipo = campos[0]

    return {
        "tipo": tipo,
        "timestamp": int(campos[1]),
        "altitude": float(campos[2]),
        "temp": float(campos[3]),
        "pressao": float(campos[4]),
        "voltage": float(campos[5]),
        "checksum": int(checksum_str, 16),
        "raw": linha,
    }

Validação de Checksum

def calcular_checksum(linha):
    # XOR de todos os bytes entre $ e *
    conteudo = linha[1:].split("*")[0]
    checksum = 0
    for char in conteudo:
        checksum ^= ord(char)
    return checksum

def validar_checksum(linha):
    _, checksum_str = linha[1:].split("*")
    esperado = int(checksum_str, 16)
    calculado = calcular_checksum(linha)
    return calculado == esperado

# Uso
if validar_checksum("$T,100,120.5,25.3*3F"):
    print("Checksum OK")
else:
    print("Dados corrompidos!")

Validação de Range e NaN

import math

LIMITES = {
    "altitude": (-500, 50000),
    "temp": (-40, 85),
    "pressao": (30000, 110000),
    "voltage": (3.0, 5.5),
}

def validar_dados(packet):
    if math.isnan(packet["altitude"]):
        return False, "altitude NaN"
    if math.isnan(packet["temp"]):
        return False, "temp NaN"

    for campo, (min_v, max_v) in LIMITES.items():
        valor = packet[campo]
        if valor < min_v or valor > max_v:
            return False, f"{campo} fora de range: {valor}"

    return True, "OK"

Pipeline de Validação

def processar_linha(linha):
    resultado = {"valido": False, "pacote": None, "erro": None}

    if not validar_checksum(linha):
        resultado["erro"] = "Checksum invalido"
        return resultado

    packet = parse_packet(linha)
    if packet is None:
        resultado["erro"] = "Formato invalido"
        return resultado

    valido, erro = validar_dados(packet)
    if not valido:
        resultado["erro"] = erro
        return resultado

    resultado["valido"] = True
    resultado["pacote"] = packet
    return resultado

Conectando com Serial

import serial

def conectar_flight_computer(porta="/dev/ttyACM0", baud=115200):
    try:
        ser = serial.Serial(
            port=porta,
            baudrate=baud,
            timeout=1,
            parity=serial.PARITY_NONE,
            stopbits=serial.STOPBITS_ONE,
        )
        print(f"Conectado a {porta} a {baud} baud")
        return ser
    except serial.SerialException as e:
        print(f"Erro ao conectar: {e}")
        return None

Leitura em Tempo Real

def ler_telemetria(ser, callback, max_linhas=100):
    contador = 0
    while contador < max_linhas:
        try:
            linha = ser.readline().decode("utf-8").strip()
            if not linha:
                continue

            resultado = processar_linha(linha)
            if resultado["valido"]:
                callback(resultado["pacote"])
                contador += 1
            else:
                print(f"Pacote rejeitado: {resultado['erro']}")

        except serial.SerialException:
            print("Conexao perdida!")
            break

def exibir_pacote(pkt):
    print(f"[{pkt['timestamp']}] Alt={pkt['altitude']:.1f}m "
          f"T={pkt['temp']:.1f}C P={pkt['pressao']:.0f}Pa")

Salvando em CSV

import csv

class LoggerCSV:
    def __init__(self, caminho="log_telemetria.csv"):
        self.arquivo = open(caminho, "w", newline="")
        self.escritor = csv.DictWriter(
            self.arquivo,
            fieldnames=["timestamp", "altitude", "temp", "pressao", "voltage"]
        )
        self.escritor.writeheader()

    def escrever(self, pacote):
        self.escritor.writerow({
            "timestamp": pacote["timestamp"],
            "altitude": pacote["altitude"],
            "temp": pacote["temp"],
            "pressao": pacote["pressao"],
            "voltage": pacote["voltage"],
        })

    def fechar(self):
        self.arquivo.close()

Script Principal

def main():
    ser = conectar_flight_computer()
    if ser is None:
        return

    logger = LoggerCSV("voo_2024.csv")

    def callback(pacote):
        exibir_pacote(pacote)
        logger.escrever(pacote)

    try:
        ler_telemetria(ser, callback, max_linhas=500)
    except KeyboardInterrupt:
        print("\nEncerrando...")
    finally:
        logger.fechar()
        ser.close()
        print("Arquivo salvo. Conexao fechada.")

if __name__ == "__main__":
    main()
