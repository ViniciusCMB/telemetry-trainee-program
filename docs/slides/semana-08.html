Projeto Capstone — Especificação

Construir um sistema completo de telemetria que:

  • Firmware (PlatformIO): lê BMP280 a 20 Hz, envia pacotes com checksum pela serial
  • FMS (Flight Mode Simulator): implementa FSM simples (IDLE → BOOST → COAST → DROGUE → LANDED)
  • Ground Station (Python): lê serial, faz parsing, valida, log CSV
  • Dashboard: gráficos em tempo real com matplotlib
  • Simulação: gerar dados sintéticos para testar sem hardware
  • Relatório: análise pós-voo automática

FMS — Exemplo de Implementação

enum State {
  IDLE, BOOST, COAST, DROGUE, MAIN, LANDED
};

State estado = IDLE;
unsigned long tempo_estado = 0;

void loop() {
  unsigned long agora = millis();

  switch (estado) {
    case IDLE:
      if (altitude > 10.0) {
        transitarPara(BOOST, agora);
      }
      break;

    case BOOST:
      if (agora - tempo_estado > 2000) {
        transitarPara(COAST, agora);
      }
      break;

    case COAST:
      if (detectarApogeu()) {
        transitarPara(DROGUE, agora);
      }
      break;

    case DROGUE:
      if (altitude < 300.0) {
        transitarPara(MAIN, agora);
      }
      break;

    case MAIN:
      if (altitude < 10.0 && abs(velocity) < 0.5) {
        transitarPara(LANDED, agora);
      }
      break;

    case LANDED:
      // beacon final, desliga transmissoes
      break;
  }
}

void transitarPara(State novo, unsigned long agora) {
  estado = novo;
  tempo_estado = agora;
  logEvento(estado);
}

Post-Flight Analysis

def analisar_pos_voo(caminho_csv):
    df = ler_csv(caminho_csv)

    print("=== ANALISE POS-VOO ===")
    print(f"Duracao: {df['timestamp'].iloc[-1] - df['timestamp'].iloc[0]:.1f}s")

    alt_max = df['altitude'].max()
    idx_max = df['altitude'].idxmax()
    print(f"Altitude maxima: {alt_max:.1f}m (t={df['timestamp'][idx_max]:.1f}s)")

    vel_max = calcular_velocidade_maxima(df)
    print(f"Velocidade maxima: {vel_max:.1f} m/s ({vel_max*3.6:.1f} km/h)")

    taxa_subida_max = calcular_taxa_subida(df)
    print(f"Taxa de subida maxima: {taxa_subida_max:.1f} m/s")
