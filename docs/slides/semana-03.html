GPIO — LED e Botão

const int LED_PIN = 13;
const int BUTTON_PIN = 2;

void setup() {
  pinMode(LED_PIN, OUTPUT);
  pinMode(BUTTON_PIN, INPUT_PULLUP);
  Serial.begin(115200);
}

void loop() {
  int estado = digitalRead(BUTTON_PIN);

  if (estado == LOW) {  // botao pressionado (pull-up)
    digitalWrite(LED_PIN, HIGH);
    Serial.println("Botao pressionado!");
  } else {
    digitalWrite(LED_PIN, LOW);
  }

  delay(50);  // debounce simples
}

Buzzer — Alertas Sonoros

const int BUZZER_PIN = 9;

void setup() {
  pinMode(BUZZER_PIN, OUTPUT);
}

void bip(int frequencia, int duracao) {
  tone(BUZZER_PIN, frequencia, duracao);
  delay(duracao);
  noTone(BUZZER_PIN);
}

void loop() {
  bip(1000, 200);  // 1 kHz por 200 ms
  delay(500);
  bip(2000, 100);  // 2 kHz por 100 ms
  delay(1000);
}

Tom agudo = maior frequência (Hz).

Serial — Pacotes Formatados

void enviar_pacote(int timestamp, float alt, float temp) {
  Serial.print("$T,");
  Serial.print(timestamp);
  Serial.print(",");
  Serial.print(alt, 2);
  Serial.print(",");
  Serial.print(temp, 2);
  Serial.print("*");

  // Calcula checksum XOR
  uint8_t cksum = 0;
  String dados = "T," + String(timestamp) + ","
                 + String(alt, 2) + "," + String(temp, 2);
  for (size_t i = 0; i < dados.length(); i++) {
    cksum ^= dados[i];
  }

  Serial.println(cksum, HEX);
}

Saída: $T,100,120.50,25.30*3F

I2C — Sensor BMP280

O BMP280 mede temperatura e pressão barométrica:

#include <Wire.h>
#include <Adafruit_BMP280.h>

Adafruit_BMP280 bmp;

void setup() {
  Serial.begin(115200);
  Wire.begin();

  if (!bmp.begin(0x76)) {  // endereco I2C padrao
    Serial.println("BMP280 nao encontrado!");
    while (1);
  }

  bmp.setSampling(Adafruit_BMP280::MODE_NORMAL);
}

Lendo o BMP280

void loop() {
  float temperatura = bmp.readTemperature();
  float pressao = bmp.readPressure() / 100.0F;  // hPa
  float altitude = bmp.readAltitude(1013.25);    // metros

  Serial.print("Temp: ");
  Serial.print(temperatura);
  Serial.print(" C, Pressao: ");
  Serial.print(pressao);
  Serial.print(" hPa, Alt: ");
  Serial.print(altitude);
  Serial.println(" m");

  delay(1000);
}

Altitude calculada a partir da pressão (fórmula barométrica).

I2C Scanner

Nem sempre sabemos o endereço do dispositivo. Um scanner resolve:

void setup() {
  Serial.begin(115200);
  Wire.begin();

  Serial.println("Scanning I2C...");
  for (byte addr = 1; addr < 127; addr++) {
    Wire.beginTransmission(addr);
    if (Wire.endTransmission() == 0) {
      Serial.print("Dispositivo em 0x");
      Serial.println(addr, HEX);
    }
  }
  Serial.println("Scan concluido.");
}

void loop() {}

Endereços comuns: BMP280 (0x76/0x77), MPU6050 (0x68), OLED (0x3C).

millis() — O Problema do delay()

delay() bloqueia o microcontrolador:

// RUIM: bloqueia todo o sistema
void loop() {
  digitalWrite(LED_PIN, HIGH);
  delay(1000);
  digitalWrite(LED_PIN, LOW);
  delay(1000);
  // Nao posso ler sensor durante o delay!
}

Solução: usar millis() para temporização não bloqueante.

millis() — Blink Sem Delay

const unsigned long INTERVALO = 1000;
unsigned long ultimo_toggle = 0;
bool estado_led = LOW;

void loop() {
  unsigned long agora = millis();

  if (agora - ultimo_toggle >= INTERVALO) {
    ultimo_toggle = agora;
    estado_led = !estado_led;
    digitalWrite(LED_PIN, estado_led);
  }

  // Posso ler sensores AQUI sem ser bloqueado!
  float temp = bmp.readTemperature();
  Serial.println(temp);
}

Taxa de Amostragem Controlada

Coletar dados a uma taxa fixa (ex: 20 Hz = 50 ms):

const unsigned long PERIODO_AMOSTRA = 50;  // 20 Hz
unsigned long ultima_amostra = 0;
int contador = 0;

void loop() {
  unsigned long agora = millis();

  if (agora - ultima_amostra >= PERIODO_AMOSTRA) {
    ultima_amostra = agora;

    float temperatura = bmp.readTemperature();
    float pressao = bmp.readPressure();

    enviar_pacote(agora, temperatura, pressao);
    contador++;
  }

  // Outras tarefas nao bloqueantes aqui
  ler_botao();
  atualizar_led_status();
}

Estrutura Completa do Firmware

#include <Wire.h>
#include <Adafruit_BMP280.h>

Adafruit_BMP280 bmp;
const unsigned long PERIODO = 50;  // 20 Hz
unsigned long ultima = 0;

void setup() {
  Serial.begin(115200);
  Wire.begin();
  bmp.begin(0x76);
}

void loop() {
  unsigned long agora = millis();
  if (agora - ultima < PERIODO) return;
  ultima = agora;

  float temp = bmp.readTemperature();
  float pressao = bmp.readPressure();
  float alt = bmp.readAltitude(1013.25);

  enviar_pacote(agora, alt, temp, pressao);
}

void enviar_pacote(unsigned long ts,
                   float alt, float temp, float pressao) {
  Serial.print("$T,");
  Serial.print(ts);
  Serial.print(",");
  Serial.print(alt, 2);
  Serial.print(",");
  Serial.print(temp, 2);
  Serial.print(",");
  Serial.print(pressao, 2);
  Serial.println("*");  // checksum omitido por brevidade
}
